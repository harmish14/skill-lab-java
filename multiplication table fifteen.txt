public class Multiplication Table FIFTEEN {
    public static void main(String args[]){
        int i=1;
        int M;


        System.out.println("Multiplication table of 15 is :");

        while(i<=10){
            M =15*i;
            System.out.println("15 x "+i +"= "+M);
            i = i+1 ;
        }
    }
}
